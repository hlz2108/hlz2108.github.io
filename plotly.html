Plotly

Problem 2

Create a flexdashboard using plotly for that includes at least three distinct plot types (e.g. scatterplots, line plots, bar plots, box plots, etc.).

library(tidyverse)
library(p8105.datasets)
library(plotly)
data("rest_inspec")

rest_inspec =
  rest_inspec %>%
  mutate(boro = str_to_title(boro)) %>% 
  filter(grade %in% c("A", "B", "C"), boro == "Manhattan") %>% 
  separate(inspection_date, into = c("year", "month", "day"), sep = "-") %>%
  mutate(
    year = as.integer(year),
    month = as.integer(month), 
    day = as.integer(day)
  ) %>% 
  relocate(boro) %>% 
  sample_n(10000) %>% 
  view

Chart A

rest_inspec %>%
  mutate(text_label = str_c("Score: ", score, "\nGrade: ", grade)) %>% 
  plot_ly(
    x = ~grade_date, y = ~score, type = "scatter", mode = "markers",
    color = ~grade, text = ~text_label, alpha = 0.5)

Chart B

rest_inspec %>%
  plot_ly(y = ~score, color = ~cuisine_description, type = "box", colors = "viridis")

Chart C

rest_inspec %>% 
  count(cuisine_description) %>% 
  mutate(cuisine_description = fct_reorder(cuisine_description, n)) %>% 
  plot_ly(x = ~cuisine_description, y = ~n, color = ~cuisine_description, type = "bar", colors = "viridis")
